C++

// ... includes ve özel fonksiyonlar ...

String myString;
int myInt;
double myDouble;
bool myBool;
int anotherInt; // DIM anotherInt AS INTEGER = 5 de buraya gelir

void setup() {
  Serial.begin(9600); // Default serial communication
  myString = "Merhaba PDSX!";
  myInt = 123;
  myDouble = 45.67;
  myBool = true;
  anotherInt = 5; // Başlangıç değeri setup'ta atanır
}

void loop() {
  Serial.print("String: ");
  Serial.println(myString);
  Serial.print("Integer: ");
  Serial.println(myInt);
  Serial.print("Double: ");
  Serial.println(myDouble);
  Serial.print("Boolean: ");
  Serial.println(myBool);

  myInt++; // ++ operatörü
  myDouble -= 10.0; // -= operatörü

  Serial.print("Integer (++): ");
  Serial.println(myInt);
  Serial.print("Double (-=): ");
  Serial.println(myDouble);

  anotherInt = (anotherInt * 3); // *= operatörü ikili işlem ve atamaya çevrilir

  Serial.print("Another Int (*=): ");
  Serial.println(anotherInt);

  Serial.print("Boolean NOT: ");
  Serial.println(!(myBool)); // NOT operatörü
  // ... sonsuz döngü devam eder, Ctrl+C ile durdurulur ...
}
